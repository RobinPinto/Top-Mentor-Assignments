#The apply() function in Pandas is a versatile method that allows you to apply a function along either the rows #or columns of a DataFrame. import pandas as pd # Sample DataFrame data = { 'A': [1, 2, 3, 4, 5], 'B': [10, 20, 30, 40, 50] } df = pd.DataFrame(data) # Define a custom function def custom_function(x): return x * 2 # Apply the custom function to a column using apply() df['A_doubled'] = df['A'].apply(custom_function) print(df)
